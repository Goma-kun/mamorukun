// askAI 関数の中の「★修正2」の部分を以下に差し替えてください async function askAI(contentsArray) { const key = localStorage.getItem('mamoru_apiKey'); if (!key) { alert('APIキーが設定されていません。右上の⚙️から設定してください。'); apiModal.classList.remove('hidden'); return 'Error: API key missing'; } const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${key}`; try { const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: contentsArray, tools: [{ googleSearch: {} }] }) }); const data = await res.json(); if (!res.ok || data.error) { const errorMsg = data.error ? data.error.message : 'Unknown API Error'; return `API Error: ${errorMsg}\n\n※GCPでのAPIキーの設定をご確認ください。`; } if (data.candidates && data.candidates[0].content.parts[0].text) { let responseText = data.candidates[0].content.parts[0].text; // 検索結果のメタデータからリンク集を作成 if (data.candidates[0].groundingMetadata && data.candidates[0].groundingMetadata.groundingChunks) { const chunks = data.candidates[0].groundingMetadata.groundingChunks; let links = []; chunks.forEach(chunk => { if (chunk.web && chunk.web.uri) { if (!links.some(l => l.uri === chunk.web.uri)) { links.push({ title: chunk.web.title || "参考ソース", uri: chunk.web.uri }); } } }); if (links.length > 0) { responseText += "\n\n**【参考リンク】**\n"; links.forEach(link => { responseText += `* [${link.title}](${link.uri})\n`; }); } } // AIが本文中に出す「内部的なリダイレクトURL」だけを消去 responseText = responseText.replace(/https:\/\/vertexaisearch\.cloud\.google\.com[^\s\)]+/g, "(リンクは下部参照)"); return responseText; } return "AIからの応答が空でした。"; } catch (err) { return "通信失敗: " + err.toString(); } }
